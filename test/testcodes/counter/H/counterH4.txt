VAR
   a;
   b;
   c;
   d;
INIT
   !a & !b & !c & !d;
TRANS
   (next(a) xor a) & (next(b) <-> (b xor (a))) & (next(c) <-> (c xor (a & b))) & (next(d) <-> (d xor (a & b & c)));
CTLSPEC
   AG(AF(a & b & c & d));
CTLSPEC
   AG(AF(!a & !b & !c & !d));
CTLSPEC
   AG((a & b & c & d) -> AX(!a & !b & !c & !d));
