VAR                 -- Declaracion de variables a utilizar
   a;               -- Bit mas significativo
   b;
   c;               
   d;
   e;
INIT 
   !a & !b & !c & !d & !e;    -- Estado inicial con todos los bits apagados
TRANS               -- Estructura de Kripke
   (next(a) xor a) & (next(b) <-> b xor a) & (next(c) <-> c xor a & b) & (next(d) <-> d xor a & b & c) & (next(e) <-> e xor a & b & c & d);
CTLSPEC
   AG(AF(!a & !b & !c & !d & !e));
CTLSPEC
   AG(a & b & c & d & e -> AX(!a & !b & !c & !d & !e));
CTLSPEC
   AG((a -> EX(!a)) | (b -> EX(!b)) | (c -> EX(!c)) | (d -> EX(!d)) | (e -> EX(!e)));
CTLSPEC
   AG((EX(a) <-> AX(a)) & (EX(b) <-> AX(b)) & (EX(c) <-> AX(c)) & (EX(d) <-> AX(d)) & (EX(e) <-> AX(e)));
