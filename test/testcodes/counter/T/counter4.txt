VAR                 -- Declaracion de variables a utilizar
   a;               -- Bit mas significativo
   b;
   c;               
   d;
INIT 
   !a & !b & !c & !d;    -- Estado inicial con todos los bits apagados
TRANS               -- Estructura de Kripke
   (next(a) xor a) & (next(b) <-> b xor a) & (next(c) <-> c xor a & b) & (next(d) <-> d xor a & b & c);
CTLSPEC
   AG(AF(!a & !b & !c & !d));
CTLSPEC
   AG(a & b & c & d -> AX(!a & !b & !c & !d));
CTLSPEC
   AG((a -> EX(!a)) | (b -> EX(!b)) | (c -> EX(!c)) | (d -> EX(!d)));
CTLSPEC
   AG((EX(a) <-> AX(a)) & (EX(b) <-> AX(b)) & (EX(c) <-> AX(c)) & (EX(d) <-> AX(d)));
