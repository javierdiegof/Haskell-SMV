VAR
    i1;
    i2;
    d1;
    d2;
    p1;
    p2;
IVAR
    s1; 
    s2;
DEFINE
    oc1 := i1 & d1 & p1;
    oc2 := i2 & d2 & p2;
    noc1 := !i1 & !d1 & !p1;
    noc2 := !i2 & !d2 & !p2;
INIT
    noc1 & noc2;
TRANS
    (s1 & noc2 -> next(oc1)) & -- si es necesario !s2
    (s2 & noc1 -> next(oc2)) & -- si es necesario !s1
    (s1 &  s2 & noc1 & noc2 -> next(oc1) xor next(oc2)) &
    (!s1 -> next(noc1)) &
    (!s2 -> next(noc2)) &
    (oc1 -> next(noc2)) &   --estan de mas
    (oc2 -> next(noc1));    -- checar si estan de mas
CTLSPEC
    (EF (p1 & p2)) | (EF (i1 & i2)) | (EF (d1 & d2));
CTLSPEC
    AG((oc1) | (noc1)) & AG((oc2) | (noc2));
