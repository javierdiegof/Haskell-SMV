VAR
    i1;
    i2;
    d1;
    d2;
    p1;
    p2;
DEFINE
    oc1 := i1 & d1 & p1;
    oc2 := i2 & d2 & p2;
    noc1 := !i1 & !d1 & !p1;
    noc2 := !i2 & !d2 & !p2;
INIT
    noc1 & noc2;
IVAR
    s1; 
    s2;
TRANS
    (s1 & noc2 -> next(oc1)) &
    (s2 & noc1 -> next(oc2)) &
    (!s1 -> next(noc1)) &
    (!s2 -> next(noc2)) &
    (p1 -> next(noc2)) &
    (p2 -> next(noc1)) &
    (s1 &  s2 & noc1 & noc2 -> next(oc1) xor next(oc2));
CTLSPEC
    !(EF (p1 & p2)) & !(EF (i1 & i2)) & !(EF (d1 & d2));
CTLSPEC
    AG((oc1) | (noc1)) & AG((oc2) | (noc2));
